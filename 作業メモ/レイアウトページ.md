# レイアウトページを作る

- ページ遷移しても一貫性のあるヘッダーとフッターを用意したい（デザインが安定すると思われる）
- レイアウトコンポーネントを作成しその中にヘッダーコンポーネントとフッターコンポーネントを含む形にする
- 各ページをレイアウトコンポーネントで囲う形で作成するようにする

# サンプルの確認

- index.js に Header と Footer コンポーネントを入れて試す

```js
// index.js
import { supabase } from 'utils/supabaseClient';
import { Auth, Button } from '@supabase/ui';
import { useEffect, useState } from 'react';
import { Profile } from 'components/Profile';
import { CompanyEdit } from 'components/CompanyEdit';
import { Header } from 'components/Header';
import { Footer } from 'components/Footer';

export default function IndexPage() {
  const { user } = Auth.useUser();
  const [prof, setProf] = useState({});
  const logout = async () => {
    const { error } = await supabase.auth.signOut();
    if (error) console.log('Error logging out:', error.message);
  };

  useEffect(() => {
    supabase
      .from('ユーザー')
      .select('*')
      .then((DB) => {
        setProf(DB.data[0]);
      });
  }, [user]);
  // console.log(prof);
  console.log(Auth.useUser());
  return (
    <>
      <Header />
      <div className="w-full h-full bg-gray-300  ">
        {!user ? (
          <div className="w-full h-full flex justify-center items-center p-4">
            <Auth
              supabaseClient={supabase}
              providers={['google', 'github']}
              socialLayout="horizontal"
              socialButtonSize="xlarge"
            />
          </div>
        ) : (
          <div
            className="w-full h-full flex flex-col justify-center items-center p-4"
            style={{ minWidth: 250, maxWidth: 600, margin: 'auto' }}
          >
            {/* <Profile logout={logout} uuid={user.id} /> */}
            <Button className="btn-black w-full mt-12" onClick={logout}>
              Logout
            </Button>
          </div>
        )}
        {/* <CompanyEdit /> */}
      </div>
      <Footer />
    </>
  );
}
```
